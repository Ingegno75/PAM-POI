<script>
  // --- Map setup ---
  const map = L.map('map', { zoomControl: true }).setView([45.07, 7.68], 10);

  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    maxZoom: 19,
    attribution: '© OpenStreetMap'
  }).addTo(map);

  let waypoints = [];
  let markers = [];

  // --- Custom marker ---
  function makeIcon(selected) {
    return L.divIcon({
      className: '',
      html: `<div style="width:14px;height:14px;border-radius:50%;background:${selected ? '#00d4ff' : '#7c3aed'};border:2px solid #fff;box-shadow:0 0 6px rgba(0,212,255,0.6)"></div>`,
      iconSize: [14, 14],
      iconAnchor: [7, 7]
    });
  }

  // --- Render ---
  function render() {
    markers.forEach(m => map.removeLayer(m));
    markers = [];

    const list = document.getElementById('list');
    list.innerHTML = '';

    if (waypoints.length === 0) {
      list.innerHTML = '<li class="empty-state">Nessun POI caricato. Cerca un indirizzo o importa un file GPX.</li>';
      document.getElementById('countLabel').textContent = '0 POI';
      return;
    }

    const sel = waypoints.filter(w => w.selected).length;
    document.getElementById('countLabel').textContent = `${sel}/${waypoints.length} selezionati`;

    waypoints.forEach((w, i) => {
      const marker = L.marker([w.lat, w.lon], { icon: makeIcon(w.selected) })
        .addTo(map)
        .bindPopup(`<b>${w.name}</b><br><small>${w.lat.toFixed(5)}, ${w.lon.toFixed(5)}</small>`);

      marker.on('click', () => {
        w.selected = !w.selected;
        render();
      });

      markers.push(marker);

      const li = document.createElement('li');
      if (w.selected) li.classList.add('selected');

      const chk = document.createElement('input');
      chk.type = 'checkbox';
      chk.checked = w.selected;
      chk.onchange = () => { w.selected = chk.checked; render(); };

      const info = document.createElement('div');
      info.className = 'poi-info';
      info.innerHTML = `<div class="poi-name">${w.name}</div><div class="poi-coords">${w.lat.toFixed(5)}, ${w.lon.toFixed(5)}</div>`;

      info.addEventListener('click', () => {
        map.setView([w.lat, w.lon], 15);
        markers[i].openPopup();
      });

      const del = document.createElement('button');
      del.className = 'btn-del';
      del.textContent = '✕';
      del.onclick = () => { waypoints.splice(i, 1); render(); };

      li.appendChild(chk);
      li.appendChild(info);
      li.appendChild(del);
      list.appendChild(li);
    });
  }

  // --- Search ---
  function doSearch() {
    const q = document.getElementById('search').value.trim();
    if (!q) return;

    fetch('https://nominatim.openstreetmap.org/search?format=json&q=' + encodeURIComponent(q), {
      headers: { 'Accept-Language': 'it' }
    })
    .then(r => r.json())
    .then(data => {
      if (!data.length) { showToast('Nessun risultato trovato'); return; }
      const r = data[0];
      const wp = {
        name: r.display_name,
        lat: parseFloat(r.lat),
        lon: parseFloat(r.lon),
        selected: true
      };
      waypoints.push(wp);
      map.setView([wp.lat, wp.lon], 13);
      render();
      showToast('POI aggiunto ✓');
    })
    .catch(() => showToast('Errore nella ricerca'));
  }

  function clearSearch() {
    document.getElementById('search').value = '';
    document.getElementById('search').focus();
  }

  function clearAll() {
    if (waypoints.length === 0) return;
    if (confirm(`Eliminare tutti i ${waypoints.length} POI?`)) {
      waypoints = [];
      render();
    }
  }

  // --- Import GPX ---
  document.getElementById('fileInput').addEventListener('change', function(e) {
    const file = e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = function() {
      const xml = new DOMParser().parseFromString(reader.result, 'text/xml');
      const pts = xml.getElementsByTagName('wpt');
      let count = 0;
      for (const p of pts) {
        const lat = parseFloat(p.getAttribute('lat'));
        const lon = parseFloat(p.getAttribute('lon'));
        const nameNode = p.getElementsByTagName('name')[0];
        waypoints.push({
          name: nameNode ? nameNode.textContent : 'Waypoint',
          lat, lon, selected: true
        });
        count++;
      }
      if (count > 0) {
        const bounds = L.latLngBounds(waypoints.map(w => [w.lat, w.lon]));
        map.fitBounds(bounds, { padding: [20, 20] });
        render();
        showToast(`${count} POI importati ✓`);
      } else {
        showToast('Nessun waypoint trovato nel file');
      }
    };
    reader.readAsText(file);
    this.value = '';
  });

  // --- Export GPX ---
  function exportGPX() {
    const selected = waypoints.filter(w => w.selected);
    if (!selected.length) { showToast('Seleziona almeno un POI'); return; }

    let xml = '<?xml version="1.0" encoding="UTF-8"?>\n';
    xml += '<gpx version="1.1" creator="PAM-POI-Manager" xmlns="http://www.topografix.com/GPX/1/1">\n';
    selected.forEach(w => {
      xml += `  <wpt lat="${w.lat}" lon="${w.lon}"><name>${escapeXml(w.name)}</name></wpt>\n`;
    });
    xml += '</gpx>';

    const blob = new Blob([xml], { type: 'application/gpx+xml' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = 'POI_export.gpx';
    a.click();
    showToast(`${selected.length} POI esportati ✓`);
  }

  function escapeXml(s) {
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }

  // --- Toast ---
  let toastTimer;
  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(() => t.classList.remove('show'), 2500);
  }

  // --- Service Worker ---
  if ('serviceWorker' in navigator) {
    navigator.serviceWorker.register('sw.js').catch(console.error);
  }

  render();
</script>
